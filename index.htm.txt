    <h2 id="header-date">📅 Отчет</h2>
    <div id="list">Loading...</div>
    <button id="main-btn" onclick="sendData()">📤 Отправить отчет</button>

    <script>
        let tg = window.Telegram.WebApp;
        tg.expand(); // На весь экран

        // 1. Получаем данные из URL (которые прислал бот)
        const urlParams = new URLSearchParams(window.location.search);
        const rawData = urlParams.get('data');
        let students = [];
        
        try {
            if (rawData) {
                // Декодируем данные
                students = JSON.parse(decodeURIComponent(rawData));
            } else {
                // Тестовые данные (если открыть просто в браузере)
                students = [
                    {id: 1, time: "14:00", name: "Тест Аня", type: "regular"},
                    {id: 2, time: "15:00", name: "Тест Саша", type: "makeup"}
                ];
            }
        } catch (e) {
            document.getElementById('list').innerHTML = "Ошибка данных: " + e;
        }

        // 2. Рисуем список
        const listDiv = document.getElementById('list');
        listDiv.innerHTML = "";

        students.forEach((stud, index) => {
            let isMakeup = stud.type === 'makeup';
            let badge = isMakeup ? '🟠 ОТРАБОТКА' : '🟢 Урок';
            
            let html = `
            <div class="student-card" id="card-${index}">
                <div class="row">
                    <div>
                        <div class="name">${stud.time} - ${stud.name}</div>
                        <div class="status">${badge}</div>
                    </div>
                    <div class="toggle-container">
                        <label>
                            <input type="checkbox" class="toggle-checkbox" id="check-${index}" onchange="toggleMakeup(${index}, ${isMakeup})">
                            <span class="toggle-slider"></span>
                        </label>
                    </div>
                </div>
                
                <div class="makeup-date" id="date-block-${index}">
                    <label>📅 За какое число долг?</label>
                    <input type="date" id="date-input-${index}">
                </div>
            </div>`;
            listDiv.innerHTML += html;
        });

        // 3. Логика показа календаря
        function toggleMakeup(index, isMakeup) {
            if (!isMakeup) return; // Если обычный урок - ничего не делаем
            
            const checkbox = document.getElementById(`check-${index}`);
            const dateBlock = document.getElementById(`date-block-${index}`);
            
            // Если галочка "Пришел" стоит -> показываем выбор даты
            if (checkbox.checked) {
                dateBlock.style.display = 'block';
            } else {
                dateBlock.style.display = 'none';
            }
        }

        // 4. Отправка данных обратно боту
        function sendData() {
            let report = [];
            let hasError = false;

            students.forEach((stud, index) => {
                let isPresent = document.getElementById(`check-${index}`).checked;
                let makeupDate = "";

                if (stud.type === 'makeup' && isPresent) {
                    let dateVal = document.getElementById(`date-input-${index}`).value;
                    if (!dateVal) {
                        alert(`❗ Укажите дату долга для: ${stud.name}`);
                        hasError = true;
                    }
                    makeupDate = dateVal; // "2024-03-01"
                }

                report.push({
                    name: stud.name,
                    time: stud.time,
                    present: isPresent,
                    is_makeup: stud.type === 'makeup',
                    makeup_for: makeupDate
                });
            });

            if (hasError) return;

            // Отправляем JSON обратно в Телеграм
            tg.sendData(JSON.stringify(report));
        }
    </script>
